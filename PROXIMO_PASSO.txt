  → ssh -T git@github.com
  → ssh-add ~/.ssh/id_ed25519
  → git push origin main

  Error: "Authentication failed"
  → gh auth login
  → git push origin main

  Error: "Rejected (non-fast-forward)"
  → git pull origin main
  → git push origin main
